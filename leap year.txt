import java.util.Scanner;
class Main{
    public static void main(String []args){
        Scanner m=new Scanner(System.in);
        System.out.println("Enter a year: ");
        int y=m.nextInt();
        if(y %4==0 || y%100!=0){
            if(y%400==0){
               System.out.println("LEAP YEAR");
            }else{
                System.out.println("NOT A LEAP YEAR");}
        }else{
            System.out.println("NOT A LEAP YEAR");
        }
    m.close();
    }
}
