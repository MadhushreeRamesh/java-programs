import java.util.Scanner;
public class Main{
    public static void main(String []args){
        Scanner a=new Scanner(System.in);
        System.out.println("Enter the number: ");
        int n=a.nextInt();
        for(int i=1;i<=10;i++){
            System.out.println(i +" * "+n+" = "+i*n);
        }
    }
}
